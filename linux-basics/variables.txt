nombre="Juan"
edad=20
Sin espacios alrededor del =

Variables especiales

$0 → nombre del script

$1, $2... → argumentos

$# → número de argumentos

$? → código de salida del último comando

$$ → PID del script

Condicionales if
if [ condicion ]; then
    comandos
elif [ otra_condicion ]; then
    comandos
else
    comandos
fi

Operadores para numeros:

-eq igual
-ne distinto
-gt mayor
-lt menor
-ge mayor o igual
-le menor o igual

Archivos:

-f archivo
-d directorio
-x ejecutable
-e Existe (archivo, directorio o enlace)
-r	Existe y tiene permiso de lectura
-w	Existe y tiene permiso de escritura
-x	Existe y es ejecutable

Strings  

-z devuelve verdadero si la cadena está vacía (longitud 0).  
-n → verdadero si tiene contenido  
= → son iguales
!= → son diferentes  

Ejemplos:
if [ $edad -ge 18 ]; then
    echo "Mayor de edad"
fi

if [ -f archivo.txt ]; then
    echo "Existe"
fi

Loops:
for i in 1 2 3; do
    echo $i
done

Con rango:
for i in {1..5}; do
  echo $i
done

Con archivos:
for archivo in *.txt; do
  echo "$archivo"
done  

contador=1
while [ $contador -le 5 ]; do
    echo $contador
    contador=$((contador+1))
done
  
until [ $contador -gt 5 ]; do
    echo $contador
    contador=$((contador+1))
done

Servicios con IF

if systemctl is-active --quiet ssh; then
    echo "Activo"
else
    echo "No activo"
fi

--quiet es para que en la salida del script no se repita el mensaje de si esta activo o no


echo "Ingrese el nombre del servicio"
read servicio

if systemctl list-unit-files | grep -q "^$servicio"; then
    if systemctl is-active --quiet "$servicio"; then
        echo "El servicio está activo"
    else
        echo "El servicio existe pero no está activo"
    fi
else
    echo "El servicio no existe"
fi
el ^ es para que busque el nombre solo al inicio del servicio, para que sea una coincidencia mas exacta
